from pythainlp.generate.wangchanglm import WangChanGLM
import torch

model = WangChanGLM()

print(model.instruct_generate(instruct="ขอวิธีลดน้ำหนัก"))

# เช่น อาหารทอด อาหารมัน อาหารที่มีน้ําตาลสูง
